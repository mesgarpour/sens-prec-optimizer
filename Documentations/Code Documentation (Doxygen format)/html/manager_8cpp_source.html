360 bool manager::filterCombByPrecision(
361  const parameters::strcStatisticalValues &stat,
362  std::vector<parameters::strcStatisticalValues> &tempStats,
363  float &delta,
364  float &tempSens,
365  float &tempPrec,
366  unsigned short &tempCombSize)
367 {
368  //if sensitivy decreasing then update the decision parameters
369  //and return true to write up the cache
370  delta = tempSens - stat.combSensitivity;
371  if(delta > EPSILON)
372  {
373  tempSens = stat.combSensitivity;
374  tempPrec = stat.combPrecision;
375  tempCombSize = stat.comb.size();
376  return false;
377  }
378  //else: similar sensitivity
379  else
380  {
381  delta = tempPrec - stat.combPrecision;
382  //if similar precision then apply the filter
383  if(delta < EPSILON && delta > -EPSILON)
384  {
385  //similar sizes then cache it
386  if(tempCombSize == stat.comb.size())
387  {
388  tempStats.push_back(stat);
389  }
390  //smaller combination size then cache it after resetting
391  else if(tempCombSize > stat.comb.size())
392  {
393  tempStats.clear();
394  tempStats.push_back(stat);
395  tempCombSize = stat.comb.size();
396  }
397  //else: Ignore larger combinations
398  return true;
399  }
400  //else: precision is decreasing then ignore
401  else
402  {
403  return false;
404  }
405  }
